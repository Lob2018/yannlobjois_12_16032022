import { useParams, Navigate } from 'react-router-dom'
import styled from 'styled-components'
import colors from '../../utils/style/colors'

import USER_MAIN_DATA from '../../mocked-data/USER_MAIN_DATA.json'
import USER_ACTIVITY from '../../mocked-data/USER_ACTIVITY.json'
import USER_AVERAGE_SESSIONS from '../../mocked-data/USER_AVERAGE_SESSIONS.json'
import USER_PERFORMANCE from '../../mocked-data/USER_PERFORMANCE.json'

import Aside from '../../components/Aside'

const DashBoardContainer = styled.div`
  margin-top: -1px;
  display: flex;
  flex-wrap: wrap;
  justify-content: flex-start;
  @media only screen and (max-width: 768px) {
  }
`

let iPositionUserMainData = 0,
  iPositionUserActivity = 0,
  iPositionUserAverageSession = 0,
  iPositionUserPerformance = 0

function isInMockedData(strId) {
  if (strId === undefined) return true
  // check if the id's string passed is a numeric value
  if (!/^\d+$/.test(strId)) return false
  // get the numeric value from the id's string passed
  const iID = parseInt(strId, 10)
  const equalId = (element) => element.id === iID
  const equalUserId = (element) => element.userId === iID
  // find the index position of the id in the mocked data
  iPositionUserMainData = USER_MAIN_DATA.findIndex(equalId)
  iPositionUserActivity = USER_ACTIVITY.findIndex(equalUserId)
  iPositionUserAverageSession = USER_AVERAGE_SESSIONS.findIndex(equalUserId)
  iPositionUserPerformance = USER_PERFORMANCE.findIndex(equalUserId)
  // check if this id is in all mocked data
  return iPositionUserMainData === -1 ||
    iPositionUserActivity === -1 ||
    iPositionUserAverageSession === -1 ||
    iPositionUserPerformance === -1
    ? false
    : true
}

/**
 * DashBoard page component
 * @component
 * @returns {React.ReactElement} The dashboard.
 */
function DashBoard() {
  const { id } = useParams()

  return isInMockedData(id) ? (
    <DashBoardContainer>
      <Aside />
      <h1>TEST</h1>
    </DashBoardContainer>
  ) : (
    <Navigate to="/*" replace={true} />
  )
}

export default DashBoard
